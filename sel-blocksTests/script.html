subtest
emit "START ${sname} ${srole}"
call doSomething sname="dilbert",srole="goof"
script doSomething sname
emit "sname=${sname},srole=${srole}"
endScript doSomething
call doSomething sname="dogbert",srole="woof"
call doSomething sname="ratbert",srole="squeak"
foreach sname "asok","alice"
call doSomething sname=sname,srole="superuser"
endForeach
emit "DONE ${sname} ${srole}"
assertEmitted "START $" + "{sname} $" + "{srole}~sname=dilbert,srole=goof~sname=dogbert,srole=woof~sname=ratbert,srole=squeak~sname=asok,srole=superuser~sname=alice,srole=superuser~DONE $" + "{sname} $" + "{srole}"
